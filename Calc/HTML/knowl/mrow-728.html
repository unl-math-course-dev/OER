\begin{align*} \int_e^{\infty} \frac{1}{x(\ln(x))^2} \, dx &= \lim_{b \to \infty} \int_e^{b} \frac{1}{x(\ln(x))^2} \, dx\\ &= \lim_{b \to \infty} \int_1^{\ln(b)} \frac{1}{u^2} \, du\\ &= \lim_{b \to \infty} \left( -u^{-1} \right)\biggm|_1^{\ln(b)}\\ &= \lim_{b \to \infty} \left( -\frac{1}{\ln(b)} - (-1) \right)\\ &= 1 \end{align*}
